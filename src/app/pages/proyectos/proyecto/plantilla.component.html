Crear Nuevo Repositorio

git clone {{dominio}}/{{proyecto.path}}.git
cd {{proyecto.nombre}}
touch README.md
git add README.md
git commit -m "add README"
git push -u origin master

Existiendo folder

cd existing_folder
git init
git remote add origin {{dominio}}/{{proyecto.path}}.git
git pull origin master
git add .
git commit -m "Initial commit"
git push -u origin master

Existiendo Git Repositorio

cd existing_repo
git remote rename origin old-origin
git remote add origin {{dominio}}/{{proyecto.path}}.git
git push -u origin --all
git push -u origin --tags
